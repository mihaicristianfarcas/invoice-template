\def\totalamount{0}

% Command to add two decimal numbers
\newcommand{\addtototal}[1]{%
    \xdef\totalamount{\fpeval{\totalamount + #1}}%
}

% Format a bare amount, padded to two decimals (75 -> 75.00)
\newcommand{\formatamount}[1]{%
    \begingroup
    \edef\fmtrounded{\fpeval{round(#1,2)}}%
    \edef\fmtinteger{\fpeval{trunc(\fmtrounded,0)}}%
    \edef\fmtcents{\fpeval{round(abs(\fmtrounded - \fmtinteger) * 100, 0)}}%
    \fmtinteger.\ifnum\fmtcents<10 0\fi\fmtcents
    \endgroup
}

% Format an amount with the currency symbol
\newcommand{\formatcurrency}[1]{%
    \formatamount{#1}\,\currencysymbol%
}
